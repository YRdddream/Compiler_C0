int gcd(int x,int y){
    int tmp;
    if(y==0)
        return (x);
    if(x<y)
    {
        tmp=x;
        x=y;
        y=tmp;
    }
    if(y!=0)
    {
        x=x-x/y*x;
        printf("counting lcf, tmp_x is ",x);
        return (gcd(y,x));
    }
}
